Scripts & Systemd Units

Complete, production-ready implementation of the LUKS + OpenBao volume encryption architecture

Section 1: luks-agent.sh

The main encryption/decryption script. Handles LUKS formatting on first boot and unlocking on subsequent boots. Supports both boot-time scanning and hot-attach via udev.

Installation path

/usr/local/bin/luks-agent.sh — Deploy via golden image or vendor data

Usage

luks-agent.sh boot          # Boot-time: scan all /dev/vd[b-z]
luks-agent.sh attach vdX    # Hot-attach: handle single device

Complete source

#!/usr/bin/env bash
# /usr/local/bin/luks-agent.sh
# LUKS volume encryption agent — manages LUKS formatting and unlocking
# using OpenBao Transit (encrypt/decrypt) and KV (ciphertext storage).
#
# Usage:
#   luks-agent.sh boot          — scan all /dev/vd[b-z] at boot
#   luks-agent.sh attach vdX    — handle single hot-attached device
#
# Requires: cryptsetup, jq, curl, lsblk

set -euo pipefail

# --- Configuration ---
CONF="/etc/luks-agent/agent.conf"
TOKEN_FILE="/run/bao-token"
VOLUMES_JSON="/etc/luks-agent/volumes.json"
IMDS_ADDR="http://169.254.169.254"
MOUNT_BASE="/mnt"
LUKS_TYPE="luks2"

# Parse agent.conf (simple ini parser)
parse_conf() {
    local key="$1"
    grep -E "^${key}\s*=" "$CONF" 2>/dev/null | sed 's/^[^=]*=\s*//' | tr -d '[:space:]'
}

BAO_ADDR=$(parse_conf "address" || echo "https://openbao.internal:8200")
TRANSIT_KEY=$(parse_conf "transit_key" || echo "luks-master")
KV_MOUNT=$(parse_conf "kv_mount" || echo "luks")

# --- Helpers ---
log() { echo "[luks-agent] $(date -Iseconds) $*" | systemd-cat -t luks-agent 2>/dev/null || echo "[luks-agent] $*"; }
die() { log "FATAL: $*"; exit 1; }

get_token() {
    [[ -f "$TOKEN_FILE" ]] || die "No BAO token at $TOKEN_FILE"
    cat "$TOKEN_FILE"
}

bao_api() {
    local method="$1" path="$2" data="${3:-}"
    local token
    token=$(get_token)
    local url="${BAO_ADDR}/v1/${path}"
    local args=(-s -f -H "X-Vault-Token: ${token}" -H "Content-Type: application/json")
    if [[ "$method" == "GET" ]]; then
        curl "${args[@]}" "$url"
    else
        curl "${args[@]}" -X "$method" -d "$data" "$url"
    fi
}

get_cinder_uuid() {
    local dev="$1"
    lsblk -ndo SERIAL "/dev/${dev}" 2>/dev/null | tr -d '[:space:]'
}

is_luks() {
    local dev="$1"
    cryptsetup isLuks "/dev/${dev}" 2>/dev/null
}

should_encrypt_boot() {
    # Check volumes.json for boot-time decision
    local uuid="$1"
    if [[ -f "$VOLUMES_JSON" ]]; then
        jq -r --arg u "$uuid" '.volumes[] | select(.uuid == $u) | .encrypt' "$VOLUMES_JSON" 2>/dev/null
    fi
}

should_encrypt_hotattach() {
    # Query IMDS for hot-attach decision
    local uuid="$1"
    local resp
    resp=$(curl -sf "${IMDS_ADDR}/v1/volumes/${uuid}" 2>/dev/null) || return 1
    echo "$resp" | jq -r '.encrypt' 2>/dev/null
}

# --- Path A: First-time encryption ---
encrypt_new_volume() {
    local dev="$1" uuid="$2"
    local mapper_name="crypt-${uuid}"
    local mount_point="${MOUNT_BASE}/${uuid}"

    log "Path A: encrypting new volume ${dev} (${uuid})"

    # 1. Generate random passphrase (memory only)
    local PASS
    PASS=$(head -c 32 /dev/urandom | base64 -w0)

    # 2. Encrypt passphrase via Transit
    local plaintext_b64
    plaintext_b64=$(echo -n "$PASS" | base64 -w0)
    local encrypt_resp
    encrypt_resp=$(bao_api POST "transit/encrypt/${TRANSIT_KEY}" \
        "{\"plaintext\": \"${plaintext_b64}\"}") || die "Transit encrypt failed for ${uuid}"
    local ciphertext
    ciphertext=$(echo "$encrypt_resp" | jq -r '.data.ciphertext')
    [[ "$ciphertext" != "null" && -n "$ciphertext" ]] || die "No ciphertext returned for ${uuid}"

    # 3. Store ciphertext in KV
    bao_api POST "${KV_MOUNT}/data/volumes/${uuid}" \
        "{\"data\": {\"ciphertext\": \"${ciphertext}\"}}" || die "KV write failed for ${uuid}"

    # 4. LUKS format
    echo -n "$PASS" | cryptsetup luksFormat --type "$LUKS_TYPE" \
        --cipher aes-xts-plain64 --key-size 512 --hash sha256 \
        --key-file - "/dev/${dev}" || die "luksFormat failed for ${dev}"

    # 5. Open, format filesystem, mount
    echo -n "$PASS" | cryptsetup open --type luks --key-file - \
        "/dev/${dev}" "$mapper_name" || die "cryptsetup open failed for ${dev}"

    mkfs.ext4 -q "/dev/mapper/${mapper_name}" || die "mkfs failed for ${mapper_name}"
    mkdir -p "$mount_point"
    mount "/dev/mapper/${mapper_name}" "$mount_point" || die "mount failed for ${mapper_name}"

    # 6. Backup LUKS header
    local header_backup
    header_backup=$(mktemp)
    cryptsetup luksHeaderBackup "/dev/${dev}" --header-backup-file "$header_backup"
    local header_b64
    header_b64=$(base64 -w0 < "$header_backup")
    bao_api POST "${KV_MOUNT}/data/headers/${uuid}" \
        "{\"data\": {\"header\": \"${header_b64}\"}}" || log "WARN: header backup failed for ${uuid}"
    rm -f "$header_backup"

    # 7. Update fstab
    if ! grep -q "$mapper_name" /etc/fstab; then
        echo "/dev/mapper/${mapper_name}  ${mount_point}  ext4  defaults,nofail  0  2" >> /etc/fstab
    fi

    # 8. Wipe passphrase
    unset PASS plaintext_b64 ciphertext

    log "Path A complete: ${mount_point} mounted (encrypted)"
}

# --- Path B: Unlock existing volume ---
unlock_existing_volume() {
    local dev="$1" uuid="$2"
    local mapper_name="crypt-${uuid}"
    local mount_point="${MOUNT_BASE}/${uuid}"

    log "Path B: unlocking existing volume ${dev} (${uuid})"

    # Already open?
    if [[ -e "/dev/mapper/${mapper_name}" ]]; then
        if mountpoint -q "$mount_point" 2>/dev/null; then
            log "Already mounted at ${mount_point}, skipping"
            return 0
        fi
        mount "/dev/mapper/${mapper_name}" "$mount_point" 2>/dev/null && {
            log "Remounted ${mount_point}"
            return 0
        }
    fi

    # 1. Get ciphertext from KV
    local kv_resp
    kv_resp=$(bao_api GET "${KV_MOUNT}/data/volumes/${uuid}") || die "KV read failed for ${uuid}"
    local ciphertext
    ciphertext=$(echo "$kv_resp" | jq -r '.data.data.ciphertext')
    [[ "$ciphertext" != "null" && -n "$ciphertext" ]] || die "No ciphertext in KV for ${uuid}"

    # 2. Decrypt via Transit
    local decrypt_resp
    decrypt_resp=$(bao_api POST "transit/decrypt/${TRANSIT_KEY}" \
        "{\"ciphertext\": \"${ciphertext}\"}") || die "Transit decrypt failed for ${uuid}"
    local PASS
    PASS=$(echo "$decrypt_resp" | jq -r '.data.plaintext' | base64 -d)
    [[ -n "$PASS" ]] || die "Empty passphrase from Transit for ${uuid}"

    # 3. Open and mount
    echo -n "$PASS" | cryptsetup open --type luks --key-file - \
        "/dev/${dev}" "$mapper_name" || die "cryptsetup open failed for ${dev}"

    mkdir -p "$mount_point"
    mount "/dev/mapper/${mapper_name}" "$mount_point" || die "mount failed for ${mapper_name}"

    # 4. Wipe passphrase
    unset PASS ciphertext

    log "Path B complete: ${mount_point} mounted (unlocked)"
}

# --- Process a single device ---
process_device() {
    local dev="$1" mode="$2"

    # Skip non-block devices and root disk
    [[ -b "/dev/${dev}" ]] || return 0
    [[ "$dev" != "vda"* ]] || return 0

    local uuid
    uuid=$(get_cinder_uuid "$dev")
    [[ -n "$uuid" ]] || { log "No Cinder UUID for ${dev}, skipping"; return 0; }

    # Check encryption flag
    local encrypt_flag
    if [[ "$mode" == "boot" ]]; then
        encrypt_flag=$(should_encrypt_boot "$uuid")
    else
        encrypt_flag=$(should_encrypt_hotattach "$uuid")
    fi

    if [[ "$encrypt_flag" != "true" ]]; then
        log "Volume ${uuid} (${dev}): encrypt=${encrypt_flag:-false}, skipping"
        return 0
    fi

    # Branch: new or existing
    if is_luks "$dev"; then
        unlock_existing_volume "$dev" "$uuid"
    else
        encrypt_new_volume "$dev" "$uuid"
    fi
}

# --- Main ---
main() {
    local mode="${1:-boot}"
    log "Starting luks-agent mode=${mode}"

    case "$mode" in
        boot)
            # Scan all virtio block devices except vda (root)
            for dev in $(lsblk -ndo NAME | grep '^vd[b-z]'); do
                process_device "$dev" "boot" || log "ERROR processing ${dev}"
            done
            ;;
        attach)
            local dev="${2:-}"
            [[ -n "$dev" ]] || die "attach mode requires device name (e.g., vdb)"
            process_device "$dev" "attach" || die "Failed processing ${dev}"
            ;;
        *)
            die "Unknown mode: ${mode}. Use 'boot' or 'attach '"
            ;;
    esac

    log "luks-agent finished mode=${mode}"
}

main "$@"
Section 2: Systemd Units

Four systemd units manage the luks-agent lifecycle: boot-time scanning, hot-attach detection, and token renewal.

luks-agent-boot.service

Runs at boot to scan all attached volumes and process them according to their encryption flags.

Installation path

/etc/systemd/system/luks-agent-boot.service

Source

[Unit]
Description=LUKS Agent — boot-time volume encryption/unlock
After=network-online.target cloud-init.target
Wants=network-online.target
ConditionPathExists=/run/bao-token
ConditionPathExists=/etc/luks-agent/volumes.json

[Service]
Type=oneshot
ExecStart=/usr/local/bin/luks-agent.sh boot
RemainAfterExit=yes
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target

luks-agent@.service

Template unit instantiated by udev when a new block device appears. Handles hot-attach volumes.

Installation path

/etc/systemd/system/luks-agent@.service

Source

[Unit]
Description=LUKS Agent — hot-attach encryption for %i
After=network-online.target
Wants=network-online.target
ConditionPathExists=/run/bao-token

[Service]
Type=oneshot
ExecStart=/usr/local/bin/luks-agent.sh attach %i
StandardOutput=journal
StandardError=journal

luks-token-renew.service

One-shot service called by the timer to renew the OpenBao token. Retrieves a new token from IMDS every 45 minutes.

Installation path

/etc/systemd/system/luks-token-renew.service

Source

[Unit]
Description=LUKS Agent — OpenBao token renewal
After=network-online.target
Wants=network-online.target

[Service]
Type=oneshot
ExecStart=/bin/bash -c '\
  RESP=$(curl -sf http://169.254.169.254/v1/renew -X POST) && \
  TOKEN=$(echo "$RESP" | jq -r ".token") && \
  echo -n "$TOKEN" > /run/bao-token && \
  chmod 0600 /run/bao-token && \
  echo "Token renewed successfully" || \
  echo "Token renewal failed, will retry"'
StandardOutput=journal
StandardError=journal

luks-token-renew.timer

Timer that triggers token renewal service every 45 minutes. Runs on boot and persists across reboots.

Installation path

/etc/systemd/system/luks-token-renew.timer

Source

[Unit]
Description=Renew OpenBao token every 45 minutes

[Timer]
OnBootSec=45min
OnUnitActiveSec=45min
AccuracySec=1min
Persistent=true

[Install]
WantedBy=timers.target
Section 3: udev Rules

udev rule that triggers the luks-agent template service whenever a new virtio block device is attached to the VM.

Installation path

/etc/udev/rules.d/99-luks-auto.rules

Source

# /etc/udev/rules.d/99-luks-auto.rules
# Trigger luks-agent on virtio block device hot-attach
#
# When a new virtio disk appears (vdb, vdc, etc.), start the
# luks-agent template service for that device.

SUBSYSTEM=="block", KERNEL=="vd[b-z]", ACTION=="add", \
  TAG+="systemd", ENV{SYSTEMD_WANTS}="luks-agent@%k.service"

---
# build-golden-image.yml
# Build a Glance image with luks-agent pre-installed
#
# Usage:
#   ansible-playbook build-golden-image.yml

- name: Build LUKS-enabled golden image
  hosts: localhost
  gather_facts: no
  vars:
    base_image: ubuntu-22.04-cloud
    output_image: ubuntu-encrypted
    flavor: m1.small
    network: build-network
    key_name: build-key
    bao_address: https://openbao.internal:8200

  tasks:
    - name: Boot temporary build VM
      openstack.cloud.server:
        name: golden-image-builder
        image: "{{ base_image }}"
        flavor: "{{ flavor }}"
        network: "{{ network }}"
        key_name: "{{ key_name }}"
        auto_ip: true
        wait: true
        timeout: 300
      register: build_vm

    - name: Wait for SSH
      wait_for:
        host: "{{ build_vm.server.public_v4 }}"
        port: 22
        delay: 10
        timeout: 120

    - name: Install packages
      delegate_to: "{{ build_vm.server.public_v4 }}"
      become: true
      apt:
        name:
          - cryptsetup
          - jq
          - curl
        state: present
        update_cache: true

    - name: Copy luks-agent.sh
      delegate_to: "{{ build_vm.server.public_v4 }}"
      become: true
      copy:
        src: files/luks-agent.sh
        dest: /usr/local/bin/luks-agent.sh
        mode: '0755'

    - name: Create config directory
      delegate_to: "{{ build_vm.server.public_v4 }}"
      become: true
      file:
        path: /etc/luks-agent
        state: directory
        mode: '0755'

    - name: Copy agent.conf
      delegate_to: "{{ build_vm.server.public_v4 }}"
      become: true
      template:
        src: templates/agent.conf.j2
        dest: /etc/luks-agent/agent.conf
        mode: '0644'

    - name: Copy systemd units
      delegate_to: "{{ build_vm.server.public_v4 }}"
      become: true
      copy:
        src: "files/{{ item }}"
        dest: "/etc/systemd/system/{{ item }}"
        mode: '0644'
      loop:
        - luks-agent-boot.service
        - luks-agent@.service
        - luks-token-renew.service
        - luks-token-renew.timer

    - name: Copy udev rules
      delegate_to: "{{ build_vm.server.public_v4 }}"
      become: true
      copy:
        src: files/99-luks-auto.rules
        dest: /etc/udev/rules.d/99-luks-auto.rules
        mode: '0644'

    - name: Enable services
      delegate_to: "{{ build_vm.server.public_v4 }}"
      become: true
      systemd:
        name: "{{ item }}"
        enabled: true
      loop:
        - luks-agent-boot.service
        - luks-token-renew.timer

    - name: Clean up for snapshot
      delegate_to: "{{ build_vm.server.public_v4 }}"
      become: true
      shell: |
        cloud-init clean --logs
        rm -rf /var/lib/cloud/instances/*
        truncate -s 0 /var/log/*.log
        history -c

    - name: Stop VM for snapshot
      openstack.cloud.server_action:
        server: golden-image-builder
        action: stop
        wait: true

    - name: Create snapshot
      openstack.cloud.image:
        name: "{{ output_image }}"
        server: golden-image-builder
        wait: true
        timeout: 600
      register: snapshot

    - name: Delete build VM
      openstack.cloud.server:
        name: golden-image-builder
        state: absent

    - name: Done
      debug:
        msg: "Golden image '{{ output_image }}' created: {{ snapshot.image.id }}"
